accuracy = []

    clf = svm.SVC(C=hyperparameter, gamma='auto')
    clf = clf.fit(train_set.iloc[:, 0:1], train_set.iloc[:, 2])
    x = clf.predict(validation.iloc[:, 0:1])
    new_accuracy = accuracy_score(validation.iloc[:, 2], x)
    if new_accuracy > best_hyperparameter:
        best_hyperparameter = new_accuracy
    accuracy.append(new_accuracy)

def transform(points):
    newPoints = points[:, :-1]
    newPoints = np.square(newPoints)
    transformation = np.array([1, 1])
    return np.column_stack((np.matmul(newPoints, transformation), points[:, -1]))
